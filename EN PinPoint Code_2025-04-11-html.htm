    <h1>PinPoint Game</h1>

    <p>Try to guess the category in as few clues as possible!</p>

 

    <!-- Hint Message -->

    <p id="hint">Hint: Types of _____________</p>

 

    <div id="clues">

        <p class="clue">Clue 1: Active</p>

        <p class="clue">Clue 2: Passive</p>

        <p class="clue">Clue 3: Outside</p>

        <p class="clue">Clue 4: Nominee</p>

        <p class="clue">Clue 5: De Facto</p>

    </div>

 

    <div>

        <input type="text" id="guess" placeholder="Enter your guess">

        <button onclick="checkGuess()">Submit Guess</button>

    </div>

 

    <p id="result"></p>

    <p id="infoMessage" style="font-weight: bold;"></p>

 

    <script>

        let currentClue = 0;

        const clues = document.querySelectorAll('.clue');

        const correctAnswer = "Types of Directors";

        let attempts = 0;

        const maxAttempts = 5;

 

        // Start the game by revealing the first clue

        window.onload = function() {

            revealClue();  // Reveal the first clue when the game starts

        }

 

        function revealClue() {

            if (currentClue < clues.length) {

                clues[currentClue].style.display = 'block';

                currentClue++;

            }

        }

 

        function checkGuess() {

            const userGuess = document.getElementById("guess").value.trim().toLowerCase();

            attempts++;

 

            if (userGuess === correctAnswer.toLowerCase()) {

                document.getElementById("result").innerText = "Correct! For more information about this topic, refer to the  page / Pour plus dinformations à ce sujet, consulter la page";

                document.getElementById("infoMessage").innerText = "";

                disableInputs();

            } else {

                // Reveal the next clue automatically if the guess is wrong

                if (currentClue < clues.length) {

                    revealClue();

                }

 

                // If max attempts are reached, show the end message

                if (attempts >= maxAttempts) {

                    document.getElementById("result").innerText = "Incorrect, the correct answer is " + correctAnswer + "!";

                    document.getElementById("infoMessage").innerText = "For more information about this topic, refer to the  page / Pour plus dinformations à ce sujet, consulter la page";

                    disableInputs();

                } else {

                    document.getElementById("result").innerText = "Try again!";

                }

            }

        }

 

        function disableInputs() {

            document.getElementById("guess").disabled = true;

            document.querySelector("button[onclick='checkGuess()']").disabled = true;

        }

    </script>
